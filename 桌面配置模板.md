# 桌面配置模板

导入前替换所有 `<PLACEHOLDER>`，再放进 Clash Verge / Mihomo。

```yaml
# Clash Verge / Mihomo desktop template
# Replace every <PLACEHOLDER> before importing.

mixed-port: 7897
allow-lan: false
mode: rule
log-level: info

proxies:
  - name: DOUBLE-HOP
    type: hysteria2
    server: <FIRST_IP>
    port: <FIRST_PORT>
    password: <HY2_PASSWORD>
    sni: <TLS_NAME>
    skip-cert-verify: true

  # Optional: local media/entertainment upstream.
  # Keep this only when another local client listens on 127.0.0.1:7890.
  - name: LOCAL-MEDIA-UPSTREAM
    type: socks5
    server: 127.0.0.1
    port: 7890
    udp: true

proxy-groups:
  - name: WORK-AI
    type: select
    proxies:
      - DOUBLE-HOP

  - name: MEDIA
    type: select
    proxies:
      - LOCAL-MEDIA-UPSTREAM
      - DIRECT

rules:
  # Media traffic can use a separate local upstream.
  - DOMAIN-SUFFIX,youtube.com,MEDIA
  - DOMAIN-SUFFIX,youtu.be,MEDIA
  - DOMAIN-SUFFIX,youtube-nocookie.com,MEDIA

  # AI/API/developer traffic uses fail-closed double hop.
```
